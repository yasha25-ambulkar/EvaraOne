const rateLimit = require('express-rate-limit');
const RedisStore = require('rate-limit-redis');
const redis = require('../config/redis'); // Requires Redis connection

// ✅ FIX #1: Graduated auth rate limiter
const authLimiter = rateLimit({
  store: new RedisStore({
    client: redis,
    prefix: 'auth-limit:'
  }),
  windowMs: 15 * 60 * 1000, // 15-minute window
  max: 5, // ✅ Max 5 attempts per 15 minutes
  message: {
    error: 'Too many authentication attempts. Please try again later.',
    retryAfter: 15 * 60 // seconds
  },
  standardHeaders: true, // Return RateLimit-* headers
  legacyHeaders: false, // Disable X-RateLimit-* headers
  keyGenerator: (req, res) => {
    // Rate limit by IP + Email (brute force protection)
    return `${req.ip}:${req.body.email || 'unknown'}`;
  },
  skip: (req, res) => {
    // ✅ Don't rate-limit superadmin or internal ips
    return req.user?.role === 'superadmin' || req.ip === '127.0.0.1';
  },
  handler: (req, res) => {
    const retryAfter = req.rateLimit.resetTime - Date.now();
    res.status(429).json({
      error: 'Too many authentication attempts',
      retryAfter: Math.ceil(retryAfter / 1000) // seconds
    });
  }
});

// ✅ FIX #2: Device list query limiter (protect expensive reads)
const queryLimiter = rateLimit({
  store: new RedisStore({
    client: redis,
    prefix: 'query-limit:'
  }),
  windowMs: 60 * 1000, // 1-minute window
  max: 30, // ✅ 30 requests per minute (down from 100)
  keyGenerator: (req, res) => {
    // Per-user limiting for authenticated endpoints
    return req.user?.uid || req.ip;
  }
});

// ✅ FIX #3: Write operation limiter (DELETE, POST protection)
const writeLimiter = rateLimit({
  store: new RedisStore({
    client: redis,
    prefix: 'write-limit:'
  }),
  windowMs: 60 * 1000, // 1-minute window
  max: 20, // ✅ 20 writes per minute
  keyGenerator: (req, res) => {
    return req.user?.uid || req.ip;
  }
});

// ✅ FIX #4: General API limiter (fallback)
const apiLimiter = rateLimit({
  windowMs: 60 * 1000,
  max: 100,
  standardHeaders: true,
  legacyHeaders: false,
  keyGenerator: (req, res) => req.user?.uid || req.ip,
  skip: (req, res) => {
    // Skip /health checks and static files
    return req.path.startsWith('/api/v1/health');
  }
});

module.exports = {
  authLimiter,
  queryLimiter,
  writeLimiter,
  apiLimiter
};

const { google } = require('googleapis');
const firestore = google.firestore('v1');
const fs = require('fs');

/**
 * Manual backup script for Firestore
 * Run: node scripts/backup-firestore.js
 * 
 * Creates a full Firestore backup and exports to Cloud Storage
 */

async function backupFirestore() {
  const projectId = process.env.FIREBASE_PROJECT_ID;
  const storageBackupBucket = process.env.BACKUP_BUCKET || 'evaraone-backups';
  
  if (!projectId) {
    throw new Error('FIREBASE_PROJECT_ID not set');
  }

  console.log(`[Backup] Starting Firestore backup for project: ${projectId}`);
  
  const auth = new google.auth.GoogleAuth({
    scopes: ['https://www.googleapis.com/auth/datastore'],
  });

  const client = await auth.getClient();

  try {
    // 1. Create backup operation
    const request = {
      parent: `projects/${projectId}`,
      requestBody: {
        outputUriPrefix: `gs://${storageBackupBucket}/firestore-backups/`,
        collectionIds: ['zones', 'customers', 'devices', 'telemetry']
      }
    };

    console.log('[Backup] Initiating export...');
    const response = await firestore.projects.databases.exportDocuments(request);
    const operationName = response.data.name;
    
    console.log(`[Backup] Export operation started: ${operationName}`);
    
    // 2. Poll operation status
    let completed = false;
    let retries = 0;
    const maxRetries = 120; // 2 hours max

    while (!completed && retries < maxRetries) {
      const opResponse = await firestore.projects.locations.operations.get({
        name: operationName
      });

      if (opResponse.data.done) {
        completed = true;
        console.log(`[Backup] ✅ Export completed successfully`);
        console.log(`[Backup] Backup location: gs://${storageBackupBucket}/firestore-backups/`);
        
        // Log backup completion
        logBackupCompletion({
          projectId,
          timestamp: new Date().toISOString(),
          bucket: storageBackupBucket,
          status: 'success',
          duration: retries * 10 + ' seconds'
        });
        
      } else if (opResponse.data.error) {
        throw new Error(`Export failed: ${opResponse.data.error.message}`);
      } else {
        retries++;
        if (retries % 6 === 0) {
          console.log(`[Backup] Still running... (${retries * 10}s elapsed)`);
        }
        await new Promise(resolve => setTimeout(resolve, 10000)); // Wait 10s
      }
    }

    if (!completed) {
      throw new Error(`Backup operation timed out after ${maxRetries * 10}s`);
    }

  } catch (error) {
    console.error('[Backup] ❌ Backup failed:', error.message);
    
    logBackupCompletion({
      projectId,
      timestamp: new Date().toISOString(),
      status: 'failure',
      error: error.message
    });
    
    process.exit(1);
  }
}

function logBackupCompletion(details) {
  const logFile = 'logs/firestore-backups.log';
  const entry = JSON.stringify({
    timestamp: new Date().toISOString(),
    ...details
  }) + '\n';
  
  fs.appendFileSync(logFile, entry);
}

// Run if called directly
if (require.main === module) {
  backupFirestore().catch(err => {
    console.error('Fatal error:', err);
    process.exit(1);
  });
}

module.exports = { backupFirestore };

const schedule = require('node-schedule');
const { backupFirestore } = require('../scripts/backup-firestore');
const logger = require('../config/pino').logger;

/**
 * Schedule automated daily Firestore backups
 */

function initializeBackupScheduler() {
  if (process.env.NODE_ENV !== 'production') {
    logger.info('[Backup] Backup scheduler disabled in development');
    return;
  }

  // ✅ Daily backup at 2 AM UTC
  schedule.scheduleJob('0 2 * * *', async () => {
    logger.info('[Backup] Starting scheduled Firestore backup...');
    
    try {
      await backupFirestore();
      logger.info('[Backup] ✅ Scheduled backup completed');
    } catch (error) {
      logger.error('[Backup] ❌ Scheduled backup failed:', error);
      // Alert ops team

        severity: 'critical',
        title: 'Firestore Backup Failed',
        message: error.message

const { db } = require('../config/firebase');
const logger = require('../utils/logger');

/**
 * Centralized Firestore listener management
 * Tracks all active listeners and ensures cleanup on disconnect
 */

class ListenerManager {
  constructor() {
    // Map: socketId -> Set of unsubscribe functions
    this.listeners = new Map();
  }

  /**
   * Register a listener tied to a socket
   * On socket disconnect, all listeners are automatically cleaned up
   */
  registerListener(socketId, unsubscribeFn) {
    if (!this.listeners.has(socketId)) {
      this.listeners.set(socketId, []);
    }
    
    this.listeners.get(socketId).push(unsubscribeFn);
    
    logger.debug(`[Listener] Registered listener for socket ${socketId.substring(0, 8)}...`);
  }

  /**
   * Cleanup all listeners for a socket
   * Call this in socket.on('disconnect')
   */
  cleanupSocket(socketId) {
    const listeners = this.listeners.get(socketId) || [];
    
    listeners.forEach((unsubscribe, index) => {
      try {
        unsubscribe();
      } catch (error) {
        logger.warn(`[Listener] Error unsubscribing listener ${index}:`, error.message);
      }
    });

    this.listeners.delete(socketId);
    
    if (listeners.length > 0) {
      logger.debug(`[Listener] Cleaned up ${listeners.length} listener(s) for socket ${socketId.substring(0, 8)}...`);
    }
  }

  /**
   * Get listener count for metrics
   */
  getStats() {
    const totalSockets = this.listeners.size;
    const totalListeners = Array.from(this.listeners.values())
      .reduce((sum, listeners) => sum + listeners.length, 0);
    
    return {
      activeSockets: totalSockets,
      totalListeners,
      averageListenersPerSocket: totalSockets > 0 
        ? (totalListeners / totalSockets).toFixed(2)
        : 0
    };
  }

  /**
   * Alert if listener count gets too high
   */
  checkHealth() {
    const stats = this.getStats();
    
    if (stats.totalListeners > 1000) {
      logger.warn('[Listener] WARNING: High listener count!', stats);
    }
    
    if (stats.averageListenersPerSocket > 10) {
      logger.warn('[Listener] WARNING: Average listeners per socket is high!', stats);
    }
    
    return stats;
  }
}

module.exports = new ListenerManager();

const gracefulShutdown = () => {
  logger.info('[Server] Received SIGTERM signal, starting graceful shutdown...');

  // Step 1: Stop accepting new connections
  server.close(() => {
    logger.info('[Server] HTTP server closed, no new connections accepted');
  });

  // Step 2: Notify all connected clients
  const sockets = io.sockets.sockets;
  logger.info(`[Server] Notifying ${sockets.size} connected sockets to disconnect...`);
  
  io.emit('server-shutting-down', {
    message: 'Server is restarting, please reconnect',
    gracePeriodSeconds: 5
  });

  // Step 3: Close all Socket.io connections
  setTimeout(() => {
    io.of('/').sockets.forEach((socket) => {
      socket.disconnect(true); // ✅ Force disconnect, triggers cleanup
    });
    logger.info('[Server] Socket.io connections closed');
  }, 1000);

  // Step 4: Close Redis connection
  if (global.redisClient) {
    global.redisClient.disconnect();
    logger.info('[Server] Redis connection closed');
  }

  // Step 5: Graceful timeout before forced exit
  const timeout = setTimeout(() => {
    logger.error('[Server] Graceful shutdown timeout, forcing exit');
    process.exit(1);
  }, 10000); // 10 second timeout

  // Clear timeout if shutdown completes
  timeout.unref();
};

process.on('SIGTERM', gracefulShutdown);

const { v4: uuidv4 } = require('uuid');
const logger = require('../utils/logger');

const requestIdMiddleware = (req, res, next) => {
  // Generate request ID if not provided
  const requestId = req.get('X-Request-ID') || uuidv4();
  req.id = requestId;
  req.requestId = requestId;

  // Set in response header
  res.set('X-Request-ID', requestId);

  // Add to all Firestore operations
  req.requestIdContext = {
    requestId,
    userId: req.user?.uid || 'anonymous',
    method: req.method,
    path: req.path,
    timestamp: new Date().toISOString()
  };

  // Log request start (debug level)
  logger.debug({
    requestId,
    method: req.method,
    path: req.path,
    ip: req.ip,
    userAgent: req.get('user-agent')
  }, 'Request started');

  // Capture response time
  const startTime = Date.now();
  res.on('finish', () => {
    const duration = Date.now() - startTime;
    logger.info({
      requestId,
      method: req.method,
      path: req.path,
      status: res.statusCode,
      durationMs: duration
    }, 'Request completed');
  });

  next();
};

module.exports = requestIdMiddleware;

  const health = {
    status: 'ok',
    timestamp: new Date().toISOString(),
    uptime: Math.floor(process.uptime()) + ' seconds',
    services: {}
  };

  // ✅ Check Firebase connectivity
  try {
    const testDoc = await db.collection('_health').doc('test').get();
    health.services.firebase = 'connected';
  } catch (error) {
    health.services.firebase = 'disconnected';
    health.status = 'degraded';
  }

  // ✅ Check Redis connectivity
  if (global.redisClient) {
    try {
      await global.redisClient.ping();
      health.services.redis = 'connected';
    } catch (error) {
      health.services.redis = 'disconnected';
      health.status = 'degraded';
    }
  } else {
    health.services.redis = 'memory_fallback';
  }

  // ✅ Check MQTT if enabled
  if (global.mqttClient) {
    health.services.mqtt = global.mqttClient.connected ? 'connected' : 'disconnected';
    if (!global.mqttClient.connected) {
      health.status = 'degraded';
    }
  } else {
    health.services.mqtt = 'disabled';
  }

  // ✅ Return appropriate status code
  const statusCode = health.status === 'ok' ? 200 : 503;
  res.status(statusCode).json(health);
});
